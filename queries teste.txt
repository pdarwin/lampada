select atleta.nome, partida.data from atleta
INNER JOIN atleta_partida on atleta.id = atleta_partida.atleta_id
INNER JOIN partida on partida.id = atleta_partida.partida_id
where atleta_partida.titular_ou_banco = "t"
GROUP BY atleta.id
